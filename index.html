function duplicateEncode(word){
    let result = '';
    word = word.toLowerCase().split('');
    for(let i = 0; i < word.length; i++){
        for(let j = 0; j < word.length; j++){
            if(i != j){
                if(word[i] == word[j]){
                    result += ')'
                    break
                }
            }
        }
        if(result.length == i + 1){
            continue
        }
        result += '('
    }
    return result
}
